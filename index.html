Criptografia Segura (AES-256 + HMAC)

Exemplo de implementação segura para proteção de dados sensíveis, utilizando criptografia simétrica com validação de integridade: 


function encryptSecure($data, $key) {
    $iv = random_bytes(openssl_cipher_iv_length('aes-256-cbc'));

    $ciphertext = openssl_encrypt(
        $data,
        'aes-256-cbc',
        $key,
        OPENSSL_RAW_DATA,
        $iv
    );

    $hmac = hash_hmac('sha256', $ciphertext, $key, true);

    return base64_encode($iv . $hmac . $ciphertext);
}

function decryptSecure($payload, $key) {
    $decoded = base64_decode($payload);

    $ivLength = openssl_cipher_iv_length('aes-256-cbc');
    $iv = substr($decoded, 0, $ivLength);
    $hmac = substr($decoded, $ivLength, 32);
    $ciphertext = substr($decoded, $ivLength + 32);

    $calculatedHmac = hash_hmac('sha256', $ciphertext, $key, true);

    if (!hash_equals($hmac, $calculatedHmac)) {
        throw new Exception('Integridade comprometida');
    }

    return openssl_decrypt(
        $ciphertext,
        'aes-256-cbc',
        $key,
        OPENSSL_RAW_DATA,
        $iv
    );
}
